```css
/* 1. 定義動畫 */
@keyframes move {
	/* 可以做多個狀態的變化 */
	0% {
      transform: translate(0, 0);
  }
  25% {
      transform: translate(1000px, 0);
  }
  50% {
      transform: translate(1000px, 500px);
  }
  75% {
      transform: translate(0, 500px);
  }
  100% {
      transform: translate(0, 0);
  }
}

div {
  width: 200px;
  height: 200px;
  background-color: pink;

	/* 2. 調用動畫 */
	animation-name: move; /* 動畫名稱 */
	animation-duration: 10s;/* 持續時間 */
}
```

```html
<body>
    <div></div>
</body>
```
